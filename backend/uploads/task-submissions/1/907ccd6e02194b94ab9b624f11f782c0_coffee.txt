    @keyword NVARCHAR(100),
    @subId INT

BEGIN

    FROM CH_Product 
    WHERE (@subId = 0 OR subcategory_id = @subId)
      AND (@keyword = '' OR name LIKE N'%' + @keyword + '%');

-- 1. Thủ tục lấy danh sách có Phân trang + Tìm kiếm + Lọc
CREATE   PROCEDURE [dbo].[get_products_paging]
    @pageIndex INT,          -- Trang hiện tại (1, 2, 3...)
    @pageSize INT,           -- Số lượng item 1 trang (VD: 10)
    @keyword NVARCHAR(100),  -- Từ khóa tìm kiếm (Tên)
    @subId INT               -- ID nhóm (0 là lấy tất cả)
AS
BEGIN
    SELECT p.*, s.name as subcategory_name 
    FROM CH_Product p
    LEFT JOIN CH_SubCategory s ON p.subcategory_id = s.id
    WHERE (@subId = 0 OR p.subcategory_id = @subId)
      AND (@keyword = '' OR p.name LIKE N'%' + @keyword + '%')
    ORDER BY p.id DESC
    OFFSET (@pageIndex - 1) * @pageSize ROWS
    FETCH NEXT @pageSize ROWS ONLY;
END
GO
